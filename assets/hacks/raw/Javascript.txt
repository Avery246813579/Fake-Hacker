package me.averydurrant.fileconverter;

public class MainFrame {
	public static void main(String[] args) {
		new MainFrame();
	}

	/*
	 * Let's create some sets for hacking
	 */
	private final String ALPHABET_SET = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz";
	private final String DIGIT_SET = "0123456789";
	private final String CHAR_SET = ALPHABET_SET + DIGIT_SET;

	public MainFrame() {
		String ip = findInternetProtocolAddress();
		System.out.println(ip);
	}

	public String findInternetProtocolAddress() {
		int[] sections = {2, 3, 2, 3};
		
		String ip = "";
		for(int section : sections){
			ip += "." + findProtocolSection(section);
		}
		
		return ip.substring(1);
	}

	public String findProtocolSection(int max){
		return findProtocolSection(0, max);
	}
	
	public String findProtocolSection(int current, int max) {
		if (max - 1 < current) {
			return "";
		}

		return reherseCharacter() + findProtocolSection(current + 1, max);
	}

	public char reherseCharacter() {
		return DIGIT_SET.charAt((int) Math.floor(Math.random() * DIGIT_SET.length()));
	}
